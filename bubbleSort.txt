// Código em C:

#include <stdio.h>

int main()
{
    int vet[10] = {0}, i, j, aux;

    printf("Metodo de ordenação BubbleSort.\n");
    /* leitura dos valores do vetor */
    for(i = 0; i < 10; i++)
    {
        printf("Digite o valor de [%d]\n", i+1);
        scanf("%d", &vet[i]);
    }
    /* ordenação do vetor */
    for(i = 9; i > -1; i--)
        for(j = 0; j <= i; j++)
            if(vet[j] > vet[j+1])
                {
                    aux = vet[j];
                    vet[j] = vet[j+1];
                    vet[j+1] = aux;
                }
    /* mostrar vetor ordenado */
    printf("Vetor ordenado: [");
    for(i = 0; i < 10; i++)
        printf("%d ", vet[i]);
    printf("]");

    return 0;
}
