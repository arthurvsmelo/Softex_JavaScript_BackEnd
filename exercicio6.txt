    escreva("Digite a nota: ")
    leia(nota)
    se (nota > maior_nota) entao
        maior_nota <- nota
        melhor_aluno <- nome

    se (nota > maior_A) entao
        maior_A <- nota
        alunoA <- nome
    fimse

        aprovados_A <- aprovados_A + 1

    fimse
